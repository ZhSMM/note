fn largest<T: PartialOrd + Copy>(list: &[T]) -> T {
    let mut larger = list[0];
    for &ele in list.iter() {
        if ele > larger {
            larger = ele;
        }
    }
    larger
}

结构体

#[derive(Debug)]
struct Point<U, V> {
    x: U,
    y: V,
}

impl<U, V> Point<U, V> {
    fn get_x(&self) -> &U {
        &self.x
    }

    fn get_y(&self) -> &V {
        &self.y
    }
}

// 调用
let point = Point { x: 12, y: 13.14 };
println!("{:#?}", point);
println!("{:?}", point);
println!(
    "Point.x = {:?}, Point.y = {:?}",
    point.get_x(),
    point.get_y()
);
